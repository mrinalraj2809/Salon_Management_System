CREATE TABLE CUSTOMER(
    F_NAME VARCHAR(8) NOT NULL,
    L_NAME VARCHAR(8),
    PHONE_NO INTEGER(12) ,
    GENDER CHAR(1) ,
    CONSTRAINT CHECK_PHONE CHECK (LENGHT(PHONE_NO)>=10 AND LENGHT(PHONE_NO)<=12),
    CONSTRAINT CHECK_GENDER CHECKcheck(GENDER in ('M', 'F', 'O')),
    CONSTRAINT PK_1 PRIMARY KEY(PHONE_NO)

);

CREATE TABLE EMPLOYEE(
    F_NAME VARCHAR(8) NOT NULL,
    L_NAME VARCHAR(8),
    PHONE_NO INTEGER(12) ,
    GENDER CHAR(1) ,
    CONSTRAINT CHECK_PHONE CHECK (LENGHT(PHONE_NO)>=10 AND LENGHT(PHONE_NO)<=12),
    CONSTRAINT CHECK_GENDER CHECKcheck(GENDER in ('M', 'F', 'O')),
    CONSTRAINT PK_1 PRIMARY KEY(PHONE_NO)

);
